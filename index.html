🔧 הגדרות Supabase

מלא פעם אחת — נשמר בדפדפן

הוראות הגדרה
  1. גש ל-supabase.com וצור פרויקט חינמי
  2. ב-Table Editor צור טבלה documents (id, title, description, pdf_url, created_at)
  3. צור טבלה signatures (id, document_id, signer_name, signature_data, signed_at)
  4. ב-Storage צור bucket בשם documents (public)
  5. העתק את ה-URL ו-anon key מ-Settings > API
  6. הדבק למעלה ולחץ שמור
SQL להרצה ב-Supabase SQL Editor:
create table documents (
  id uuid default gen_random_uuid() primary key,
  title text not null,
  description text,
  pdf_url text,
  created_at timestamptz default now()
);

create table signatures (
  id uuid default gen_random_uuid() primary key,
  document_id uuid references documents(id),
  signer_name text not null,
  signature_data text not null,
  signed_at timestamptz default now()
);

-- Allow public read/insert
alter table documents enable row level security;
alter table signatures enable row level security;
create policy "public read docs" on documents for select using (true);
create policy "public insert sigs" on signatures for insert with check (true);
create policy "public read sigs" on signatures for select using (true);
